80/// Options controlling the traffic simulation.
81#[derive(Clone, StructOpt)]
82pub struct SimOptions {
83    /// Used to distinguish savestates for running the same scenario.
84    #[structopt(long, default_value = "unnamed")]
85    pub run_name: String,
86    /// Ignore all stop signs and traffic signals, instead using a "freeform" policy to control
87    /// access to intersections. If a requested turn doesn't conflict with an already accepted one,
88    /// immediately accept it. FIFO ordering, no balancing between different movements.
89    #[structopt(long)]
90    pub use_freeform_policy_everywhere: bool,
91    /// Allow a vehicle to start a turn, even if their target lane is already full. This may mean
92    /// they'll get stuck blocking the intersection.
93    #[structopt(long)]
94    pub allow_block_the_box: bool,
95    /// Normally as a vehicle follows a route, it opportunistically make small changes to use a different lane,
96    /// based on some score of "least-loaded" lane. Disable this default behavior.
97    #[structopt(long)]
98    pub dont_recalc_lanechanging: bool,
99    /// Normally if a cycle of vehicles depending on each other to turn is detected, temporarily allow
100    /// "blocking the box" to try to break gridlock. Disable this default behavior.
101    #[structopt(long)]
102    pub dont_break_turn_conflict_cycles: bool,
103    /// Disable experimental handling for "uber-turns", sequences of turns through complex
104    /// intersections with short roads. "Locks" the entire movement before starting, and ignores
105    /// red lights after starting.
106    #[structopt(long)]
107    pub dont_handle_uber_turns: bool,
108    /// Enable an experimental SEIR pandemic model. This requires an RNG seed, which can be the
109    /// same or different from the one used for the rest of the simulation.
110    #[structopt(long, parse(try_from_str = parse_rng))]
111    pub enable_pandemic_model: Option<XorShiftRng>,
112    /// When a warning is encountered during simulation, specifies how to respond.
113    #[structopt(long, parse(try_from_str = parse_alert_handler), default_value = "print")]
114    pub alerts: AlertHandler,
115    /// Ignore parking data in the map and instead treat every building as if it has unlimited
116    /// capacity for vehicles.
117    ///
118    /// Some maps always have this hardcoded on -- see the code for the list.
119    #[structopt(long)]
120    pub infinite_parking: bool,
121    /// Allow all agents to immediately proceed into an intersection, even if they'd hit another
122    /// agent. Obviously this destroys realism of the simulation, but can be used to debug
123    /// gridlock. Also implies freeform_policy, so vehicles ignore traffic signals.
124    #[structopt(long)]
125    pub disable_turn_conflicts: bool,
126    /// Don't collect any analytics. Only useful for benchmarking and debugging gridlock more
127    /// quickly.
128    #[structopt(long)]
129    pub skip_analytics: bool,
130}

161#[derive(Clone)]
162pub enum AlertHandler {
163    /// Just print the alert to STDOUT
164    Print,
165    /// Print the alert to STDOUT and don't proceed until the UI calls clear_alerts()
166    Block,
167    /// Don't do anything
168    Silence,
169}
170
171impl Default for AlertHandler {
172    fn default() -> AlertHandler {
173        AlertHandler::Print
174    }
175}
176
177fn parse_alert_handler(x: &str) -> Result<AlertHandler> {
178    match x {
179        "print" => Ok(AlertHandler::Print),
180        "block" => Ok(AlertHandler::Block),
181        "silence" => Ok(AlertHandler::Silence),
182        _ => bail!("Bad --alerts={}. Must be print|block|silence", x),
183    }
184}

232    pub(crate) fn spawn_trips(
233        &mut self,
234        input: Vec<(PersonID, TripInfo, StartTripArgs)>,
235        map: &Map,
236        timer: &mut Timer,
237    ) {
238        timer.start_iter("spawn trips", input.len());
239        for (p, info, args) in input {
240            timer.next();
241
242            let trip = self.trips.new_trip(p, info.clone());
243            // This might be immediately true due to ScenarioModifiers
244            if let Some(msg) = info.cancellation_reason {
245                self.trips.cancel_unstarted_trip(trip, msg);
246            } else {
247                self.scheduler
248                    .push(info.departure, Command::StartTrip(trip, args));
249            }
250        }
251
252        if let Some(ref mut m) = self.pandemic {
253            m.initialize(self.trips.get_all_people(), &mut self.scheduler);
254        }
255
256        self.dispatch_events(Vec::new(), map);
257    }

337    fn start_bus(&mut self, route: &TransitRoute, map: &Map) {
338        // Spawn one bus for the first leg.
339        let path = self.transit.create_empty_route(route, map);
340
341        // For now, no desire for randomness. Caller can pass in list of specs if that ever
342        // changes.
343        let (vehicle_type, length) = match route.route_type {
344            PathConstraints::Bus => (VehicleType::Bus, BUS_LENGTH),
345            PathConstraints::Train => (VehicleType::Train, LIGHT_RAIL_LENGTH),
346            _ => unreachable!(),
347        };
348        let vehicle = VehicleSpec {
349            vehicle_type,
350            length,
351            max_speed: None,
352        }
353        .make(
354            CarID {
355                id: self.trips.new_car_id(),
356                vehicle_type,
357            },
358            None,
359        );
360
361        self.scheduler.push(
362            self.time,
363            Command::SpawnCar(
364                CreateCar {
365                    router: Router::follow_bus_route(vehicle.id, path),
366                    vehicle,
367                    maybe_parked_car: None,
368                    trip_and_person: None,
369                    maybe_route: Some(route.id),
370                },
371                true,
372            ),
373        );
374    }

386impl Sim {
387    // Advances time as minimally as possible, also limited by max_dt. Returns true if the callback
388    // said to halt the sim.
389    fn minimal_step(
390        &mut self,
391        map: &Map,
392        max_dt: Duration,
393        maybe_cb: &mut Option<Box<dyn SimCallback>>,
394    ) -> bool {
395        self.step_count += 1;
396
397        let max_time = if let Some(t) = self.scheduler.peek_next_time() {
398            if t > self.time + max_dt {
399                // Next event is after when we want to stop.
400                self.time += max_dt;
401                return false;
402            }
403            t
404        } else {
405            // No events left at all
406            self.time += max_dt;
407            return false;
408        };
409
410        let mut halt = false;
411        while let Some(time) = self.scheduler.peek_next_time() {
412            if time > max_time {
413                return false;
414            }
415            if let Some(cmd) = self.scheduler.get_next() {
416                if self.do_step(map, time, cmd, maybe_cb) {
417                    halt = true;
418                    break;
419                }
420            }
421        }
422
423        halt
424    }
425
426    // If true, halt simulation because the callback said so.
427    fn do_step(
428        &mut self,
429        map: &Map,
430        time: Time,
431        cmd: Command,
432        maybe_cb: &mut Option<Box<dyn SimCallback>>,
433    ) -> bool {
434        self.time = time;
435        let mut events = Vec::new();
436        let mut halt = false;
437
438        let mut ctx = Ctx {
439            parking: &mut self.parking,
440            intersections: &mut self.intersections,
441            scheduler: &mut self.scheduler,
442            map,
443            handling_live_edits: None,
444        };
445
446        match cmd {
447            Command::StartTrip(id, args) => {
448                self.trips.start_trip(self.time, id, args, &mut ctx);
449            }
450            Command::SpawnCar(create_car, retry_if_no_room) => {
451                // If this SpawnCar is being retried and the map was live-edited since the first
452                // attempt, the path might've become invalid. TODO Skip this check
453                // most of the time.
454                let constraints = create_car.vehicle.vehicle_type.to_constraints();
455                let mut ok = true;
456                for step in create_car.router.get_path().get_steps() {
457                    match step.as_traversable() {
458                        Traversable::Lane(l) => {
459                            if !constraints.can_use(ctx.map.get_l(l), ctx.map) {
460                                ok = false;
461                                break;
462                            }
463                        }
464                        Traversable::Turn(t) => {
465                            if ctx.map.maybe_get_t(t).is_none() {
466                                ok = false;
467                                break;
468                            }
469                        }
470                    }
471                }
472                if !ok {
473                    self.trips.cancel_trip(
474                        self.time,
475                        create_car.trip_and_person.unwrap().0,
476                        "path is no longer valid after map edits".to_string(),
477                        Some(create_car.vehicle),
478                        &mut ctx,
479                    );
480                } else {
481                    // create_car contains a Path, which is expensive to clone. We need different
482                    // parts of create_car after attempting start_car_on_lane.
483                    // Make copies just of those here. In no case do we ever
484                    // clone the path.
485                    let id = create_car.vehicle.id;
486                    let maybe_route = create_car.maybe_route;
487                    let trip_and_person = create_car.trip_and_person;
488                    let maybe_parked_car = create_car.maybe_parked_car.clone();
489                    let req = create_car.router.get_path().get_req().clone();
490
491                    if let Some(create_car) = self
492                        .driving
493                        .start_car_on_lane(self.time, create_car, &mut ctx)
494                    {
495                        // Starting the car failed for some reason.
496                        if retry_if_no_room {
497                            // Although the agent isn't on the map yet, they're trying.
498                            if let Some((trip, _)) = trip_and_person {
499                                self.trips.agent_starting_trip_leg(AgentID::Car(id), trip);
500                            }
501                            self.driving.vehicle_waiting_to_spawn(
502                                id,
503                                req.start,
504                                trip_and_person.map(|(_, p)| p),
505                            );
506
507                            // TODO Record this in the trip log
508                            self.scheduler.push(
509                                self.time + BLIND_RETRY_TO_SPAWN,
510                                Command::SpawnCar(create_car, retry_if_no_room),
511                            );
512                        } else if let Some((trip, person)) = create_car.trip_and_person {
513                            self.trips.cancel_trip(
514                                self.time,
515                                trip,
516                                format!(
517                                    "no room to spawn car for {} by {}, not retrying",
518                                    trip, person
519                                ),
520                                Some(create_car.vehicle),
521                                &mut ctx,
522                            );
523                        }
524                    } else {
525                        // Creating the car succeeded.
526                        if let Some((trip, person)) = trip_and_person {
527                            self.trips.agent_starting_trip_leg(AgentID::Car(id), trip);
528                            events.push(Event::TripPhaseStarting(
529                                trip,
530                                person,
531                                Some(req),
532                                if id.vehicle_type == VehicleType::Car {
533                                    TripPhaseType::Driving
534                                } else {
535                                    TripPhaseType::Biking
536                                },
537                            ));
538                        }
539                        if let Some(parked_car) = maybe_parked_car {
540                            if let ParkingSpot::Offstreet(b, _) = parked_car.spot {
541                                // Buses don't start in parking garages, so trip must exist
542                                events.push(Event::PersonLeavesBuilding(
543                                    trip_and_person.unwrap().1,
544                                    b,
545                                ));
546                            }
547                            self.parking.remove_parked_car(parked_car);
548                        }
549                        if let Some(route) = maybe_route {
550                            self.transit.bus_created(id, route);
551                        }
552                        self.analytics
553                            .record_demand(self.driving.get_path(id).unwrap(), map);
554                    }
555                }
556            }
557            Command::SpawnPed(create_ped) => {
558                // Do the order a bit backwards so we don't have to clone the
559                // CreatePedestrian. spawn_ped can't fail.
560                self.trips
561                    .agent_starting_trip_leg(AgentID::Pedestrian(create_ped.id), create_ped.trip);
562                events.push(Event::TripPhaseStarting(
563                    create_ped.trip,
564                    create_ped.person,
565                    Some(create_ped.path.get_req().clone()),
566                    TripPhaseType::Walking,
567                ));
568                self.analytics.record_demand(&create_ped.path, map);
569
570                // Maybe there's actually no work to do!
571                match (&create_ped.start.connection, &create_ped.goal.connection) {
572                    (
573                        SidewalkPOI::Building(b1),
574                        SidewalkPOI::ParkingSpot(ParkingSpot::Offstreet(b2, idx)),
575                    ) if b1 == b2 => {
576                        self.trips.ped_reached_parking_spot(
577                            self.time,
578                            create_ped.id,
579                            ParkingSpot::Offstreet(*b2, *idx),
580                            Duration::ZERO,
581                            Distance::ZERO,
582                            &mut ctx,
583                        );
584                    }
585                    _ => {
586                        if let SidewalkPOI::Building(b) = &create_ped.start.connection {
587                            events.push(Event::PersonLeavesBuilding(create_ped.person, *b));
588                        }
589
590                        self.walking
591                            .spawn_ped(self.time, create_ped, map, &mut self.scheduler);
592                    }
593                }
594            }
595            Command::UpdateCar(car) => {
596                self.driving.update_car(
597                    car,
598                    self.time,
599                    &mut ctx,
600                    &mut self.trips,
601                    &mut self.transit,
602                    &mut self.walking,
603                );
604            }
605            Command::UpdateLaggyHead(car) => {
606                self.driving.update_laggy_head(car, self.time, &mut ctx);
607            }
608            Command::UpdatePed(ped) => {
609                self.walking.update_ped(
610                    ped,
611                    self.time,
612                    &mut ctx,
613                    &mut self.trips,
614                    &mut self.transit,
615                );
616            }
617            Command::UpdateIntersection(i) => {
618                self.intersections
619                    .update_intersection(self.time, i, map, &mut self.scheduler);
620            }
621            Command::Callback(frequency) => {
622                self.scheduler
623                    .push(self.time + frequency, Command::Callback(frequency));
624                if maybe_cb.as_mut().unwrap().run(self, map) {
625                    halt = true;
626                }
627            }
628            Command::Pandemic(cmd) => {
629                self.pandemic
630                    .as_mut()
631                    .unwrap()
632                    .handle_cmd(self.time, cmd, &mut self.scheduler);
633            }
634            Command::StartBus(r, _) => {
635                self.start_bus(map.get_tr(r), map);
636            }
637        }
638
639        // Record events at precisely the time they occur.
640        self.dispatch_events(events, map);
641
642        halt
643    }

645    fn dispatch_events(&mut self, mut events: Vec<Event>, map: &Map) {
646        events.extend(self.trips.collect_events());
647        events.extend(self.transit.collect_events());
648        events.extend(self.driving.collect_events());
649        events.extend(self.walking.collect_events());
650        events.extend(self.intersections.collect_events());
651        events.extend(self.parking.collect_events());
652        for ev in events {
653            if let Some(ref mut m) = self.pandemic {
654                m.handle_event(self.time, &ev, &mut self.scheduler);
655            }
656            if let Some(ref mut r) = self.recorder {
657                r.handle_event(self.time, &ev, map, &self.driving);
658            }
659
660            self.analytics.event(ev, self.time, map);
661        }
662    }

664    pub fn timed_step(
665        &mut self,
666        map: &Map,
667        dt: Duration,
668        maybe_cb: &mut Option<Box<dyn SimCallback>>,
669        timer: &mut Timer,
670    ) {
671        let end_time = self.time + dt;
672        let start = Instant::now();
673        let mut last_update = Instant::now();
674
675        timer.start(format!("Advance sim to {}", end_time));
676        while self.time < end_time {
677            if self.minimal_step(map, end_time - self.time, maybe_cb) {
678                break;
679            }
680            if !self.analytics.alerts.is_empty() {
681                match self.alerts {
682                    AlertHandler::Print => {
683                        for (t, loc, msg) in self.analytics.alerts.drain(..) {
684                            println!("Alert at {} ({:?}): {}", t, loc, msg);
685                        }
686                    }
687                    AlertHandler::Block => {
688                        for (t, loc, msg) in &self.analytics.alerts {
689                            println!("Alert at {} ({:?}): {}", t, loc, msg);
690                        }
691                        break;
692                    }
693                    AlertHandler::Silence => {
694                        self.analytics.alerts.clear();
695                    }
696                }
697            }
698            if Duration::realtime_elapsed(last_update) >= Duration::seconds(1.0) {
699                // TODO Not timer?
700                println!(
701                    "- After {}, the sim is at {}. {} live agents",
702                    Duration::realtime_elapsed(start),
703                    self.time,
704                    prettyprint_usize(self.num_active_agents()),
705                );
706                last_update = Instant::now();
707            }
708        }
709        timer.stop(format!("Advance sim to {}", end_time));
710    }
711    pub fn tiny_step(&mut self, map: &Map, maybe_cb: &mut Option<Box<dyn SimCallback>>) {
712        self.timed_step(
713            map,
714            Duration::seconds(0.1),
715            maybe_cb,

722        map: &Map,
723        dt: Duration,

839    /// Respond to arbitrary map edits without resetting the simulation. Returns the number of
840    /// (trips cancelled, parked cars displaced).
841    pub fn handle_live_edits(&mut self, map: &Map, timer: &mut Timer) -> (usize, usize) {
842        self.edits_name = map.get_edits().edits_name.clone();
843
844        let (affected, num_parked_cars) = self.find_trips_affected_by_live_edits(map, timer);
845        let num_trips_cancelled = affected.len();
846        let affected_agents: BTreeSet<AgentID> = affected.iter().map(|(a, _)| *a).collect();
847
848        // V1: Just cancel every trip crossing an affected area.
849        // (V2 is probably rerouting everyone, only cancelling when that fails)
850        // TODO If we delete a bus, deal with all its passengers
851        let mut ctx = Ctx {
852            parking: &mut self.parking,
853            intersections: &mut self.intersections,
854            scheduler: &mut self.scheduler,
855            map,
856            handling_live_edits: Some(affected_agents),
857        };
858        for (agent, trip) in affected {
859            match agent {
860                AgentID::Car(car) => {
861                    let vehicle = self.driving.delete_car(car, self.time, &mut ctx);
862                    // TODO Plumb more info about the reason
863                    self.trips.cancel_trip(
864                        self.time,
865                        trip,
866                        "map edited without reset".to_string(),
867                        Some(vehicle),
868                        &mut ctx,
869                    );
870                    self.trips.trip_abruptly_cancelled(trip, AgentID::Car(car));
871                }
872                AgentID::Pedestrian(ped) => {
873                    self.walking.delete_ped(ped, &mut ctx);
874                    self.trips.cancel_trip(
875                        self.time,
876                        trip,
877                        "map edited without reset".to_string(),
878                        None,
879                        &mut ctx,
880                    );
881                    self.trips
882                        .trip_abruptly_cancelled(trip, AgentID::Pedestrian(ped));
883                }
884                AgentID::BusPassenger(_, _) => unreachable!(),
885            }
886        }
887
888        self.driving.handle_live_edits(map);
889        self.intersections.handle_live_edits(map);
890
891        (num_trips_cancelled, num_parked_cars)
892    }
893
894    /// Returns (trips affected, number of parked cars displaced)
895    fn find_trips_affected_by_live_edits(
896        &mut self,
897        map: &Map,
898        timer: &mut Timer,
899    ) -> (BTreeSet<(AgentID, TripID)>, usize) {
900        let mut affected: BTreeSet<(AgentID, TripID)> = BTreeSet::new();
901
902        // TODO Handle changes to access restrictions
903
904        {
905            // Find every active trip whose path crosses a modified lane or intersection
906            let (edited_lanes, _) = map.get_edits().changed_lanes(map);
907            let mut closed_intersections = HashSet::new();
908            for i in map.get_edits().original_intersections.keys() {
909                if map.get_i(*i).is_closed() {
910                    closed_intersections.insert(*i);
911                }
912            }
913            for (a, trip) in self.trips.active_agents_and_trips() {
914                if let Some(path) = self.get_path(*a) {
915                    if path
916                        .get_steps()
917                        .iter()
918                        .any(|step| match step.as_traversable() {
919                            Traversable::Lane(l) => edited_lanes.contains(&l),
920                            Traversable::Turn(t) => {
921                                closed_intersections.contains(&t.parent)
922                                    || edited_lanes.contains(&t.src)
923                                    || edited_lanes.contains(&t.dst)
924                            }
925                        })
926                    {
927                        affected.insert((*a, *trip));
928                    }
929                }
930            }
931
932            affected.extend(
933                self.driving
934                    .find_vehicles_affected_by_live_edits(&closed_intersections, &edited_lanes),
935            );
936        }
937
938        let num_evicted = {
939            let (evicted_cars, cars_parking_in_the_void) =
940                self.parking.handle_live_edits(map, timer);
941            let num_evicted = evicted_cars.len();
942            affected.extend(self.walking.find_trips_to_parking(evicted_cars));
943            for car in cars_parking_in_the_void {
944                let a = AgentID::Car(car);
945                affected.insert((a, self.agent_to_trip(a).unwrap()));
946            }
947
948            if !self.parking.is_infinite() {
949                let (filled, avail) = self.parking.get_all_parking_spots();
950                let mut all_spots: BTreeSet<ParkingSpot> = BTreeSet::new();
951                all_spots.extend(filled);
952                all_spots.extend(avail);
953                affected.extend(self.driving.find_trips_to_edited_parking(all_spots));
954            }
955            num_evicted
956        };
957
958        (affected, num_evicted)
959    }
