12pub fn run(mut settings: Settings) {
13    let mut opts = map_gui::options::Options::load_or_default();
14    opts.show_building_driveways = false;
15    let args = map_gui::SimpleAppArgs::from_iter(abstutil::cli_args());
16    args.override_options(&mut opts);
17
18    settings = args
19        .update_widgetry_settings(settings)
20        .canvas_settings(opts.canvas_settings.clone());
21    widgetry::run(settings, |ctx| {
22        map_gui::SimpleApp::new(
23            ctx,
24            opts,
25            Some(args.map_name()),
26            args.cam,
27            (),
28            |ctx, app| vec![viewer::Viewer::new_state(ctx, app)],
29        )
30    });
31}
