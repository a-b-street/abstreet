38    pub fn fill(area_type: AreaType, cs: &ColorScheme) -> Fill {
39        match area_type {
40            AreaType::Park => cs.grass.clone(),
41            AreaType::Water => cs.water.clone(),
42            AreaType::Island => cs.map_background.clone(),
43            AreaType::StudyArea => cs.study_area.clone(),
44        }
45    }
