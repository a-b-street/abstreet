11pub fn run(csv_path: String, map: String) -> Result<()> {
12    let mut timer = Timer::new("import grid2demand");
13    timer.start("parse CSV");
14    let people = parse_trips(csv_path)?;
15    timer.stop("parse CSV");
16    let map = Map::load_synchronously(map, &mut timer);
17
18    let mut s = Scenario::empty(&map, "grid2demand");
19    // Include all buses/trains
20    s.only_seed_buses = None;
21    let orig_num = people.len();
22    let skip_problems = true;
23    s.people = ExternalPerson::import(&map, people, skip_problems)?;
24    // Always clean up people with no-op trips (going between the same buildings)
25    s = s.remove_weird_schedules(true);
26    println!(
27        "Imported {}/{} people",
28        prettyprint_usize(s.people.len()),
29        prettyprint_usize(orig_num)
30    );
31    s.save();
32
33    Ok(())
34}

36fn parse_trips(csv_path: String) -> Result<Vec<ExternalPerson>> {
37    let mut people = Vec::new();
38    for rec in csv::Reader::from_reader(fs_err::File::open(csv_path)?).deserialize() {
39        let rec: Record = rec?;
40        let mode = match rec.agent_type.as_ref() {
41            "v" => TripMode::Drive,
42            "b" => TripMode::Bike,
43            "p" => TripMode::Walk,
44            "t" => TripMode::Transit,
45            x => bail!("Unknown agent_type {}", x),
46        };
47        let (origin, destination) = parse_linestring(&rec.geometry)
48            .ok_or_else(|| anyhow!("didn't parse geometry {}", rec.geometry))?;
49        let departure = parse_time(rec.departure_time)?;
50        // For each row in the CSV file, create a person who takes a single trip from the origin to
51        // the destination. They do not take a later trip to return home.
52        people.push(ExternalPerson {
53            trips: vec![ExternalTrip {
54                departure,
55                origin: ExternalTripEndpoint::Position(origin),
56                destination: ExternalTripEndpoint::Position(destination),
57                mode,
58                purpose: TripPurpose::Work,
59            }],
60        });
61    }
62    Ok(people)
63}

65fn parse_linestring(input: &str) -> Option<(LonLat, LonLat)> {
66    // Input is something like LINESTRING(-122.3245062 47.6456213,-122.3142029 47.6675654)
67    let mut nums = Vec::new();
68    for x in input
69        .strip_prefix("LINESTRING(")?
70        .strip_suffix(')')?
71        .split(&[' ', ','][..])
72    {
73        nums.push(x.parse::<f64>().ok()?);
74    }
75    if nums.len() != 4 {
76        return None;
77    }
78    Some((LonLat::new(nums[0], nums[1]), LonLat::new(nums[2], nums[3])))
79}

81fn parse_time(input: String) -> Result<Time> {
82    // Input is HHMM, like 0730
83    let hours = input[0..2].parse::<usize>()?;
84    let mins = input[2..].parse::<usize>()?;
85    Ok(Time::START_OF_DAY + Duration::hours(hours) + Duration::minutes(mins))
86}
